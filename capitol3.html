Verificarea numerelor prime

Subprogramul verifică dacă un număr este prim. 


#include <iostream>
using namespace std;

bool prim(int n)
{
    if(n < 2)
        return false;

    for(int d = 2; d * d <= n; d++)
    {
        if(n % d == 0)
            return false;
    }

    return true;
}

int main()
{
    if(prim(17))
        cout << "Numar prim";
    else
        cout << "Nu este prim";

    return 0;
}

Calcularea CMMDC

Algoritmul lui Euclid folosind funcții. 


#include <iostream>
using namespace std;

int cmmdc(int a, int b)
{
    while(b != 0)
    {
        int r = a % b;

        a = b;
        b = r;
    }

    return a;
}

int main()
{
    cout << cmmdc(24, 36);

    return 0;
}

AICI vei putea introduce problema ta personală.

Exemplu: Se citește un număr natural n. Să se afișeze suma cifrelor pare.

Rezolvare C++


#include <iostream>
using namespace std;

int sumaPare(int n)
{
    int s = 0;

    while(n > 0)
    {
        int c = n % 10;

        if(c % 2 == 0)
            s += c;

        n /= 10;
    }

    return s;
}

int main()
{
    int n;

    cin >> n;

    cout << sumaPare(n);

    return 0;
}
